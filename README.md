## Install

```bash
npm install --save credit-card-react
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'credit-card-react'
import 'credit-card-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
